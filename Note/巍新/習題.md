```java
public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);

    int number1 = sc.nextInt();  // 把第一個輸入的數字存入number1
    String operation = sc.next();  // 把輸入的計算符號存入字串operation
    int number2 = sc.nextInt();  // 把第二個輸入的數字存入number2

    switch (operation){  // 條件判斷(operation為+ - * / 的時候)
        case "+":
            System.out.println(number1 + " + " + number2 + " = " + (number1 + number2));
            break;
        case "-":
            System.out.println(number1 + " - " + number2 + " = " + (number1 - number2));
            break;
        case "*":
            System.out.println(number1 + " * " + number2 + " = " + (number1 * number2));
            break;
        case  "/":
            System.out.println(number1 + " / " + number2 + " = " + (number1 / number2));
            break;
    }
}
```    
