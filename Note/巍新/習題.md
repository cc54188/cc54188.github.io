### 類別練習
```java
package demo.model;

public class MyApp {

    public static void main(String[] args) {
        Car myCar = new Car(10);  // 建構時傳入速度
        System.out.println(myCar.getName());  // 取得名字
        System.out.println("速度: " + myCar.getSpeed());
        System.out.println("移動: " + myCar.move(10));

        Trunk myTrunk = new Trunk(15);
        System.out.println(myTrunk.getName());
        System.out.println("速度: " + myTrunk.getSpeed());
        System.out.println("移動: " + myTrunk.move(10));

        Racing myRacing = new Racing(20, "法拉利");
        System.out.println(myRacing.getName());
        System.out.println("速度: " + myRacing.getSpeed());
        System.out.println("移動: " + myRacing.move(10));
        
        System.out.println(myRacing instanceof Car);  // 印出跑車實體是否繼承自Car類別(true)
    }
}
```

```java
package demo.model;

public class Car {
    // 屬性皆設私有，外面若要取得，另設方法
    private String name = "車子";
    private int wheel;
    private int speed;
    private int weight = 20;

    // 設兩個建構子
    public Car(int speed) {  // 建構時傳入速度
        this.wheel = wheel = 4;  // 輪胎固定為4個
        this.speed = speed;  // 速度初始值由外面建構時決定
    }

    public Car(int speed, String name) {  // 建構時傳入速度，名稱
        this(speed);  // 直接先呼叫只傳入速度的建構子(會初始輪胎，速度)
        this.name = name;   // 初始名稱設為外面傳入值
    }

    /**
     * 移動
     */
    public int move(int time) {  // 傳入時間
        return time * speed;    // 回傳移動距離
    }

    /**
     * 讓外面取得輪胎數
     * @return
     */
    public int getWheel() {
        return wheel;
    }
    /**
     * 讓外面取得速度
     * @return
     */
    public int getSpeed() {
        return speed;
    }
    /**
     * 讓外面取得名字
     */
    public String getName() {
        return this.name;
    }

    /**
     * 是否用光油
     * @return
     */
    public boolean isOutOfGas() {
        return false;
    }

    /**
     * 加多少油
     * @return
     */
    public int getSreFlashOil() {
        return 0;
    }
}
```

```java
package demo.model;

public class Racing extends Car{
    private String name = "跑車";  // 自己定義name遮蔽父類別的name

    // 繼承父類別的兩個建構子
    public Racing(int speed) {
        super(speed);
    }

    public Racing(int speed, String name) {
        super(speed, name);
    }
}
```

```java
package demo.model;

public class Trunk extends Car{
    private String name = "卡車";

    // 只繼承父類別的一個建構子，所以外面無法用兩個參數的建構子來建構卡車
    public Trunk(int speed) {
        super(speed);
    }

    @Override
    public String getName() {  // 由於name不是用建構子傳入值，所以要覆寫方法才能回傳卡車的name
        return name;
    }
}
```

public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);

    int number1 = sc.nextInt();  // 把第一個輸入的數字存入number1
    String operation = sc.next();  // 把輸入的計算符號存入字串operation
    int number2 = sc.nextInt();  // 把第二個輸入的數字存入number2

    switch (operation){  // 條件判斷(operation為+ - * / 的時候)
        case "+":
            System.out.println(number1 + " + " + number2 + " = " + (number1 + number2));
            break;
        case "-":
            System.out.println(number1 + " - " + number2 + " = " + (number1 - number2));
            break;
        case "*":
            System.out.println(number1 + " * " + number2 + " = " + (number1 * number2));
            break;
        case  "/":
            System.out.println(number1 + " / " + number2 + " = " + (number1 / number2));
            break;
    }
}
